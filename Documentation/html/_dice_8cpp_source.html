53Roll Dice::parseRoll(std::string str) {
54 // Initialize part of the string being parsed to 1 and numDice/die to empty strings.
55 int part {1};
56 bool plus = false;
57 std::string numDice {};
58 std::string die {};
59 // numAdded initialized to '0' as no 'z' in roll string xdy[+z] means adding 0 to dice roll.
60 std::string numAdded {'0'};
61
62 // Loops through string indices, changing parts when 'd' or '+' are found
63 // and appending found digits to end of their respective strings.
64 for (int i {0}; i <= str.length(); i++) {
65 if (str[i] == 'd') {
66 part = 2;
67 continue;
68 }
69 else if (str[i] == '+') {
70 if (plus && !checkIfNumber(str[i])) {
71 diceRollStringErr(str);
72 }
73 plus = true;
74 part = 3;
75 continue;
76 }
77
78 if (part == 1 && str[i] >= '0' && str[i] <= '9') {
79 numDice += str[i];
80 }
81 else if (part == 2 && str[i] >= '0' && str[i] <= '9') {
82 die += str[i];
83 }
84 else if (part == 3 && str[i] >= '0' && str[i] <= '9') {
85 numAdded += str[i];
86 }else if (i == str.length()) {
87 break;
88 }
89 }
90
91 // Calls Dice::parseDie method to convert die string to a DieType
92 DieType dieType {parseDie(die, str)};
93
94 // Attempt to convert string for roll to ints, handling errors.
95 try {
96 return Roll{dieType, stoi(numDice), stoi(numAdded)};
97 } catch(std::invalid_argument &e) {
98 diceRollStringErr(str);
99 }

108int Dice::roll(std::string rollString) {
109 // Calls Dice::parseRoll to parse string into a roll usable with Random::random()
110 Roll roll_guide {Dice::parseRoll(rollString)};
111
112 // Initialize total to 0
113 int total {0};
114
115 // Iterates number of times equal to the number of dice to roll (x).
116 for (int i {0}; i < roll_guide.numDice; i++) {
117 // Each individual roll is rolled randomly between 1 and the max for the DieType (y)
118 // (ex: d4 max (y) = 4, d100 max (y) = 100), and adds each roll to total
119 total += Random::random(1, roll_guide.dieType);
120 }
