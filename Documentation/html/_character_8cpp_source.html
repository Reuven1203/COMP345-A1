139int Character::initializeHitPoints() {

141 int baseHP = getDieType() + getAbilityModifier(Constitution);
142 std::string levelString = std::to_string(level-1);
143 return baseHP + dice.roll(levelString + 'd' + std::to_string(getDieType())+ '+' + std::to_string(getAbilityModifier(Constitution) * (level-1)));
